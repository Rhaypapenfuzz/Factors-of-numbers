// void that displays factors. List used in void
void numFactors(int userInteger) 
{ 
list<int> factors;
//int factors[]={};
int factor; 

for (int i = 1; i <= userInteger; i++)
  if (userInteger % i==0) 
      factors.push_back(i); 
for (auto v : factors) 
    cout << v << "\n";
}

//Void that outputs factors. No list used
void numFactors2(int userInteger) 
{  
for (int i = 1; i <= userInteger; i++)
  if (userInteger % i==0) 
    cout<<i<<" ";
}
